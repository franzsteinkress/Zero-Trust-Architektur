15/**
16 * @brief Startet den mTLS-Client-Prozess.
17 * * Baut eine gesicherte Verbindung zum Server auf, führt den Handshake durch
18 * und sendet verschlüsselte Sensordaten.
19 * * @param target_host Der DNS-Name des Zielservers (wichtig für den SAN-Check).
20 * @return asio::awaitable<void> Coroutine-Handle für die asynchrone Ausführung.
21 */
22asio::awaitable<void> run_mtls_client(std::string target_host) {
23 auto executor = co_await asio::this_coro::executor;
24
25 try {
26 // 1. SSL Kontext für mTLS konfigurieren
27 ssl::context ctx(ssl::context::tls_client);
28 ctx.set_verify_mode(ssl::verify_peer);
29 ctx.load_verify_file(get_cert_path("ca.pem"));
30 ctx.use_certificate_chain_file(get_cert_path("client_cert.pem"));
31 ctx.use_private_key_file(get_cert_path("client_key.pem"), ssl::context::pem);
32
33 // 2. Verbindung aufbauen
34 ssl::stream<tcp::socket> stream(executor, ctx);
35 tcp::resolver resolver(executor);
36 log_msg("CLIENT", "INFO", std::format("Verbinde zu {} auf Port 8443...", target_host));
37 auto endpoints = co_await resolver.async_resolve(target_host, "8443", asio::use_awaitable);
38 co_await asio::async_connect(stream.next_layer(), endpoints, asio::use_awaitable);
39
40 // 3. TLS Handshake
41 log_msg("CLIENT", "INFO", "Starte mTLS Handshake...");
42 co_await stream.async_handshake(ssl::stream_base::client, asio::use_awaitable);
43 log_msg("CLIENT", "INFO", "ERFOLG: Handshake abgeschlossen. Verbindung ist sicher.");
44
45 // 4. Sicherer Datenaustausch
46 // 4.1 Text-Anfrage senden (wie bisher)
47 std::string request = "POST /sensor_update HTTP/1.1\r\n"
48 "Host: " + target_host + "\r\n"
49 "Content-Type: application/octet-stream\r\n"
50 "Connection: close\r\n\r\n";
51 co_await asio::async_write(stream, asio::buffer(request), asio::use_awaitable);
52
53 // 4.2 Binäre Sensor-Payload senden (Automotive-Demo)
54 SensorPayload sensor_data{ 42, 22.5f, static_cast<uint64_t>(std::time(nullptr)) };
55 co_await asio::async_write(stream, asio::buffer(&sensor_data, sizeof(sensor_data)), asio::use_awaitable);
56 log_msg("CLIENT", "INFO", "HTTP-Anfrage und binäre Sensor-Payload gesendet.");
57
58 // 5. Antwort empfangen
59 char data[1024];
60 size_t n = co_await stream.async_read_some(asio::buffer(data), asio::use_awaitable);
61 log_msg("CLIENT", "INFO", std::format("--- Server Antwort Start ---"));
62 log_msg("CLIENT", "INFO", std::format("{}", std::string_view(data, n)));
63 log_msg("CLIENT", "INFO", std::format("--- Server Antwort Ende ---"));
64 } catch (const std::exception& e) {
65 log_msg("CLIENT", "ERROR", std::format("CLIENT-FEHLER: {}", e.what()));
66 }
67}
